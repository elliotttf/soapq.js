1 /**
  2  * @fileoverview Class definition for processing soapq requests.
  3  */
  4 
  5 var util = require('util'),
  6     http = require('http'),
  7     https = require('https'),
  8     url = require('url'),
  9     mongoose = require('mongoose'),
 10     requests = require('../models/models.js').requests;
 11 
 12 // Event variables.
 13 var events = require('events');
 14 
 15 /**
 16  * Base class constructor.
 17  *
 18  * @param {string} key
 19  *   The request key that will be stored in the database.
 20  * @param {array} payload
 21  *   The payload to send to the remote SOAP server.
 22  * @param {string} respondURL
 23  *   The location to return the SOAP result to.
 24  * @constructor
 25  * @extends {events.EventEmitter}
 26  */
 27 function SOAPQ(key, payload, respondURL) {
 28   // Initialize the database connection.
 29   this.db = mongoose.connect('mongodb://localhost/soapq');
 30   this.Model = mongoose.model('requests', requests);
 31 
 32   // Class variables.
 33   this.key = key;
 34   this.payload = [];
 35   this.respondURL = null;
 36   this.responses = [];
 37   this.requestStatus = true;
 38 
 39   if (typeof payload != 'undefined') {
 40     this.payload = payload;
 41   }
 42 
 43   if (typeof respondURL != 'undefined') {
 44     this.respondURL = respondURL;
 45   }
 46 
 47   events.EventEmitter.call(this);
 48 }
 49 util.inherits(SOAPQ, events.EventEmitter);
 50 
 51 // Class constants.
 52 /**
 53  * Maximum number of times to retry a failed request.
 54  * @const
 55  * @type {int}
 56  */
 57 SOAPQ.RETRY_THRESHOLD = 3;
 58 
 59 /**
 60  * Adds the soapq request to the database.
 61  */
 62 SOAPQ.prototype.save = function() {
 63   var self = this;
 64 
 65   // Save the value to the database.
 66   var req = new self.Model();
 67   req.key = self.key;
 68   req.payload = self.payload;
 69   req.respondURL = self.respondURL;
 70   req.save(function(err, reqSaved) {
 71     if (err) {
 72       console.log(err);
 73       throw 'Error saving request';
 74     }
 75 
 76     self.emit('savedRequest', 'Saved request to database (' + self.key + ')');
 77   });
 78 };
 79 
 80 /**
 81  * Loads an existing soapq request from the database.
 82  */
 83 SOAPQ.prototype.load = function() {
 84   var self = this;
 85   self.Model.findOne({ 'key': self.key }, function(err, doc) {
 86     if (err) {
 87       console.log(err);
 88       throw 'Error loading request';
 89     }
 90 
 91     self.payload = doc.payload;
 92     self.respondURL = doc.respondURL;
 93   });
 94 };
 95 
 96 /**
 97  * Removes the soapq request from the database.
 98  */
 99 SOAPQ.prototype.remove = function() {
100   var self = this;
101   self.Model.remove({ 'key': self.key }, function(err) {
102     if (err) {
103       console.log(err);
104       throw 'Error removing document';
105     }
106     self.emit('removedRequest', 'Removed request from database (' + self.key + ')');
107   });
108 };
109 
110 /**
111  * Sends the soapq request to the remote server.
112  */
113 SOAPQ.prototype.request = function() {
114   var self = this;
115   var count = 0;
116   var retry = 0;
117 
118   // Request all of this payload's requests one after the other.
119   var payload = self.payload[count];
120   self.requestOne(payload);
121   self.on('requestedOne', function(response) {
122     // Reset the retry count.
123     retry = 0;
124 
125     // Store the response XML.
126     self.responses.push(response);
127 
128     // If we're done requesting, notify.
129     if (self.responses.length == self.payload.length) {
130       self.emit('requestedRequests', 'All requests have been proccessed (' + self.key + ')');
131       return;
132     }
133 
134     // Increment the counter and make the next request.
135     count++;
136     if (count < self.payload.length) {
137       self.requestOne(payload[count]);
138     }
139   });
140 
141   // If a request failed and we haven't retried more than what we decide
142   // is sane, retry the failed request.
143   self.on('requestError', function(message) {
144     if (retry < self.RETRY_THRESHOLD) {
145       console.log('Retrying request (' + self.key + ')');
146       retry++;
147       self.requestOne(payload[count]);
148     }
149     else {
150       console.log('Too many retry failures, bailing (' + self.key + ')');
151       self.emit('requestErrors', 'There was a problem communicating with the server (' + self.key + ')');
152     }
153   });
154 
155   // After all requests have been processed, respond.
156   self.on('requestedRequests', function(message) {
157     console.log(message);
158     self.respond();
159   });
160 
161   self.on('requestErrors', function(message) {
162     self.requestStatus = false;
163     self.respond();
164   });
165 };
166 
167 /**
168  * Sends a single request to the SOAP server.
169  *
170  * @param {object} payload
171  *   An object containing the following:
172  *    <ul>
173  *      <li>endpoint - the SOAP server to contact.</li>
174  *      <li>envelope - the SOAP envelope to send.</li>
175  *      <li>security (optional) - the SOAP server httpauth credentials.</li>
176  *    </ul>
177  */
178 SOAPQ.prototype.requestOne = function(payload) {
179   var self = this;
180   var endpoint = url.parse(payload.endpoint);
181   var options = {
182     host: endpoint.host,
183     path: endpoint.pathname,
184     method: 'POST',
185     headers: {
186       'Content-Type': 'text/xml',
187       'Content-Length': payload.envelope.length
188     }
189   };
190 
191   if (typeof payload.security != 'undefined') {
192     options.auth = payload.security.username + ':' + payload.security.password;
193   }
194 
195   // Send along the stuff!
196   console.log('Sending request to ' + payload.endpoint);
197   if (endpoint.protocol == 'https:') {
198     var req = https.request(options, function(res) {
199       var response = '';
200       res.on('data', function(chunk) {
201         response += chunk;
202       });
203 
204       res.on('end', function() {
205         self.emit('requestedOne', response);
206       });
207     });
208   }
209   else {
210     var req = http.request(options, function(res) {
211       var response = '';
212       res.on('data', function(chunk) {
213         response += chunk;
214       });
215 
216       res.on('end', function() {
217         // TODO - verify that this is the only valid return code.
218         if (res.statusCode != 200) {
219           self.emit('requestError', 'Error during request (' + self.key + ')');
220         }
221         else {
222           self.emit('requestedOne', response);
223         }
224       });
225     });
226   }
227 
228   try {
229     req.on('error', function(e) {
230       console.log('There was a problem processing request with key: ' + doc.key + ' and endpoint: ' + payload.endpoint);
231       self.emit('requestError', 'Error during request (' + self.key + ')');
232     });
233     req.write(payload.envelope);
234     req.end();
235   }
236   catch (e) {
237     console.log('There was a problem connecting to the API endpoint: ' + payload.endpoint);
238     self.emit('requestError', 'Error during request (' + self.key + ')');
239   }
240 };

242 /**
243  * Responds to the respond callback with the SOAP result.
244  */
245 SOAPQ.prototype.respond = function() {
246   var self = this;
247 
248   var response_json = {
249     'status': self.requestStatus,
250     'responses': self.responses
251   };
252   response_json = JSON.stringify(response_json);
253 
254   // Respond to the respond with the result.
255   console.log('Sending response to: ' + self.respondURL);
256 
257   var respond = url.parse(self.respondURL);
258   var options = {
259     'host': respond.host,
260     'path': respond.path,
261     'method': 'POST',
262     'headers': {
263       'Content-Type': 'application/json',
264       'Content-Length': response_json.length
265     }
266   };
267 
268   var req = http.request(options, function(res) {
269     var client_response = '';
270     res.on('data', function(chunk) {
271       client_response += chunk;
272     });
273 
274     res.on('end', function() {
275       client_response = JSON.parse(client_response);
276       if (client_response.status == true) {
277         console.log('Sent response to: ' + self.respondURL + ' server responded with: ' + client_response.data);
278         self.remove();
279         self.emit('processedRequest');
280       }
281       else {
282         console.log('There was a problem sending the response to: ' + self.respondURL);
283       }
284     });
285   });
286 
287   try {
288     req.on('error', function(e) {
289       console.log('There was a problem calling back: ' + self.respondURL + ' with the result.');
290     });
291     req.write(response_json);
292     req.end();
293   }
294   catch (e) {
295     console.log('There was a problem connecting to the respond server: ' + self.respondURL);
296   }
297 };
