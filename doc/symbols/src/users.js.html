  5 var events = require('events');
  6 var mongoose = require('mongoose');
  7 var users = require('../models/models.js').users;
  8 var util = require('util');
  9 var uuid = require('node-uuid');
 10 
 11 /**
 12  * Base user class.
 13  *
 14  * @param {string} email
 15  *   The user's email address.
 16  * @param {string} domain
 17  *   The domain the API will be used from.
 18  * @param {string} key
 19  *   (optional) The API key for this user. If this value
 20  *   is passed in email and domain will be ignored and
 21  *   the user will be loaded from the database.
 22  * @constructor
 23  * @extends {events.EventEmitter}
 24  */
 25 function SoapQUser(email, domain, key) {
 26   var self = this;
 27   // Database info.
 28   self.db = mongoose.connect('mongodb://localhost/soapq');
 29   self.Users = mongoose.model('users', users);
 30 
 31   mongoose.connection.on('error', function dbError(err) {
 32     self.emit('dbError', 'Error connecting to database');
 33   });
 34 
 35   // Class variables.
 36   self.email = email;
 37   self.domain = domain;
 38   self.key = null;
 39   self.active = true;
 40   self.created = self.updated = Date.now();
 41 
 42  // Load the user from the database if we have a key.
 43   if (typeof key !== 'undefined') {
 44     self.Users.findOne({ 'key': key }, function foundOne(err, doc) {
 45       if (err) {
 46         self.emit('error', err);
 47         return;
 48       }
 49       if (doc === null) {
 50         self.emit('error', 'User not found.');
 51         return;
 52       }
 53 
 54       self.key = doc.key;
 55       self.email = doc.email;
 56       self.domain = doc.domain;
 57       self.active = doc.active;
 58       self.created = doc.created;
 59       self.updated = doc.updated;
 60 
 61       self.emit('loaded', 'User data loaded from database.');
 62     });
 63   }
 64 }
 65 util.inherits(SoapQUser, events.EventEmitter);
 66 
 67 /**
 68  * Validates new user information before saving.
 69  */
 70 SoapQUser.prototype.validate = function() {
 71   var self = this;
 72   // Check the database for the email address.
 73   self.Users.findOne({ 'email': self.email }, function foundOne(err, doc) {
 74     if (err) {
 75       console.log(err);
 76 
 77       self.emit('error', 'Validation error');
 78       self.emit('validatedWithErrors', 'Validation complete, with errors.');
 79       return;
 80     }
 81     if (doc !== null) {
 82       self.emit('error', 'Mail in use');
 83       self.emit('validatedWithErrors', 'Validation complete, with errors.');
 84       return;
 85     }
 86 
 87     self.emit('validated', 'Validation complete.');
 88   });
 89 };
 90 
 91 /**
 92  * Saves the user in the database.
 93  */
 94 SoapQUser.prototype.save = function() {
 95   var self = this;
 96   var user = new self.Users();
 97   user.email = self.email;
 98   user.domain = self.domain;
 99   user.active = self.active;
100   user.key = uuid.v1();
101   user.created = self.created;
102   user.updated = self.updated;
103   user.save(function onSave(err, doc) {
104     if (err) {
105       console.log(err);
106       self.emit('error', err);
107       return;
108     }
109     console.log('New API key! (' + doc.key + ')');
110     self.key = doc.key;
111     self.emit('saved', doc.key);
112   });
113 };
114 
115 /**
116  * Deletes the user from the database.
117  */
118 SoapQUser.prototype.remove = function() {
119   var self = this;
120   if (typeof self.key === 'undefined') {
121     self.emit('error', 'Cannot remove user with unknown key.');
122     return;
123   }
124   self.Users.remove({ 'key': self.key }, function removed(err) {
125     if (err) {
126       console.log(err);
127       self.emit(err);
128       return;
129     }
130     console.log('Removed user. (' + self.key + ')');
131     self.emit('removed', 'User removed successfully');
132   });
133 };
134 
135 /**
136  * Authenticates the API request.
137  */
138 SoapQUser.prototype.authenticate = function() {
139   var self = this;
140   self.Users.findOne({ 'key': self.key }, function foundOne(err, doc) {
141     if (err) {
142       console.log(err);
143       self.emit('errorAuthenticating', err);
144     }
145 
146     // Not found, bail.
147     if (doc === null) {
148       self.emit('errorAuthenticating', 'Key not found.');
149       return;
150     }
151 
152     if (typeof doc.active === 'undefined' || !doc.active) {
153       self.emit('errorAuthenticating', 'This key is inactive.');
154       return;
155     }
156 
157     self.emit('authenticated', 'Found active key.');
158   });
159 };

161 /**
162  * Exports the SoapQUser class.
163  */
164 exports.SoapQUser = SoapQUser;
